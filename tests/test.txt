Function factorial (n as Integer, b as integer) As Integer

End Function
Scope 'Hlavni telo programu
    Dim a As Integer
    a = factorial(2, 3)
End Scope

/'vystup by mal byt (medzery neriesim, nemusia tam byt)
    .IFJcode17
    JUMP l_main
LABEL factorial
    DEFVAR TF@n
    POPS TF@n
    DEFVAR TF@b
    POPS TF@b

    DEFVAR TF@tmp
    MOVE TF@tmp int@0

    DEFVAR TF@$E_E0
    MOVE TF@$E_E0 int@5
    PUSHS TF@$E_E0
    POPS TF@tmp

    DEFVAR TF@$E_E2
    MOVE TF@$E_E2 TF@tmp
    PUSHS TF@$E_E2
    POPS TF@%RETVAL

    RETURN
LABEL l_main
    CREATEFRAME

    DEFVAR TF@a
    MOVE TF@a int@0

    DEFVAR TF@$E_E4	#parameter1
    MOVE TF@$E_E4 int@2
    DEFVAR TF@$E_E5	#parameter2
    MOVE TF@$E_E5 int@3
    PUSHS TF@$E_E5	#push od posledneho
    PUSHS TF@$E_E4
    DEFVAR TF@$E_E6	#navrat
    PUSHFRAME
    CREATEFRAME
    DEFVAR TF@%RETVAL
    CALL factorial
    MOVE LF@$E_E6 TF@%RETVAL
    POPFRAME

    PUSHS TF@$E_E6
    POPS TF@a

    WRITE TF@a  #debug
'/
